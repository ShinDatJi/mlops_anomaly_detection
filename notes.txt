uv package manager:
install: curl -LsSf https://astral.sh/uv/install.sh | sh
init new project: uv init apps/APP --no-workspace
add library: uv add --directory apps/APP lib==1.2.0
run python script: uv run apps/APP/src/SCRIPT.py
sync: uv sync --directory apps/APP
run project script: CATEGORY=bottle uv run --env-file apps/APP/.env --project apps/APP PROJECT_SCRIPT

clean docker desktop disk space:
wsl --shutdown
Optimize-VHD -Path "C:\Users\<YOU>\AppData\Local\Docker\wsl\disk\docker_data.vhdx" -Mode Full
